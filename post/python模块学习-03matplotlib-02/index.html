1. 对数坐标轴 plt.semilogx()

绘制对数坐标图的函数有三个

  • plt.semilogx() X轴为对数坐标
  • plt.semilogy() Y轴为对数坐标
  • plt.loglog() 两个轴都为对数坐标
# *-* coding:utf8 *-*
import numpy as np
import matplotlib.pyplot as plt

w = np.linspace(0.1, 1000, 1000)
p = np.abs(1/(1+0.1j*w)) # 计算低通滤波器的频率响应

plt.subplot(221)
plt.plot(w, p, linewidth=2)
plt.ylim(0, 1.5)

plt.subplot(222)
plt.semilogx(w, p, linewidth=2)
plt.ylim(0, 1.5)

plt.subplot(223)
plt.semilogy(w, p, linewidth=2)
plt.ylim(0,1.5)

plt.subplot(224)
plt.loglog(w, p, linewidth=2)
plt.ylim(0,1.5)

plt.show()

2. 极坐标图 plt.polar()

  • plt.polar()

  • plt.subplot(polar=True)

  • 程序中调用subplot()创建子图时通过设 polar参数为True,创建一个极坐标子图。

  • 然后调用plot()在极坐标子图中绘图。

  • 也可以使用polar()直接创建极坐标子图并在其中绘制曲线

# *-* coding:utf8 *-*
# 极坐标系中的点由一个夹角和一段相对中心点的距离来表示
import numpy as np
import matplotlib.pyplot as plt

theta = np.arange(0, 2*np.pi, 0.02)
# plt.figure(1)的作用是创建一个绘图纸
# 需要一个新的绘图纸就在代码上面加一句
plt.figure(1)
# 把创建的绘图纸用subplot分成几块
plt.subplot(121,polar=True)
# 用plot在分好的那块上面画图
plt.plot(theta,1.6*np.ones_like(theta),linewidth=2)
plt.plot(3*theta, theta/3,"--", linewidth=2)

# 子图2
plt.subplot(122, polar=True)
plt.plot(theta, 1.4*np.cos(5*theta), "--", linewidth=2)
plt.plot(theta, 1.8*np.cos(4*theta), linewidth=2)
# rgrids()设置同心圆栅格的半径大小和文字标注的角度
# 因此图中的虚线圆圈有三个，
# 半径分别为0.5、1.0和1.5,这些文字沿着45°线排列
plt.rgrids(np.arange(0.5, 2, 0.5), angle=45)
# Thetagrids()设置放射线栅格的角度
# 因此图中只有两条放射线，角度分别为0°和45°。 
plt.thetagrids([0, 45])


plt.show()
plt.savefig("polar.png",dpi=120)

import matplotlib.pyplot as plt

plt.polar()
plt.show()

3. 柱状图 plt.bar()

  • 第一个参数为每根柱子左边缘的横坐标，
  • 第二个参数为每根柱子的高度，
  • 第三个参数指定所有柱子的宽度。
  • 当第三个参数为序列时，可以为每根柱子指定宽度。
# *-* coding:utf8 *-*
import numpy as np
import matplotlib.pyplot as plt

# 生成一个文件
# 第一列年龄；第二列男性人数；第三列女性人数
data = np.zeros([10,3])
data[:,0] = np.arange(0,100,10)
data[:,1] = np.random.randint(0,1000,size=10)
data[:,2] = np.random.randint(0,1000,size=10)
np.savetxt("china_population.txt", data, fmt="%d", delimiter=" ")

# 读入文件的参数 dtype=float,
bar_data = np.loadtxt("china_population.txt", delimiter=" ")
width = (data[1, 0] - data[0, 0])*0.4

plt.figure(1, figsize=(8,5))
# Bar()的
# #第一个参数为每根柱子左边缘的横坐标，
# 第二个参数为每根柱子的高度，
# 第三个参数指定所有柱子的宽度。
# 当第三个参数为序列时，可以为每根柱子指定宽度。
plt.bar(data[:,0],data[:,1], width, color="lightblue",label="man")
plt.bar(data[:,0]+ width, data[:,2], width, color="red",label="woman")

plt.xlabel("number")
plt.xlim(0,90)
plt.xticks(fontsize=10, color="black", rotation=20)
plt.legend()

plt.figure(2,figsize=(8,4))
n = 10
plt.subplot(121)
plt.plot(np.random.uniform(0.5,1.0,50),'o',color="green")
normal_x = np.arange(n)
normal_dat = np.random.uniform(0.5,1.0,n)

plt.subplot(122)
# y1放在上面，y2放在下面
# #显示颜色facecolor,边缘颜色：edgecolor
# 添加柱状图数字，ha显示在柱状图上的位置,va:数字显示在柱状图上面，还是下面。

plt.bar(normal_x, normal_dat,0.5, facecolor="red", edgecolor="white")
plt.bar(normal_x, -normal_dat,0.5, facecolor="blue", edgecolor="white")

for x,y in zip(normal_x, normal_dat):
    plt.text(x, y, "%.2f" %y,ha="center", va="bottom")

for x,y in zip(normal_x, -normal_dat):
    plt.text(x, y, "%.2f" %y,ha="center", va="top")

plt.show()

4. 散点图 plt.scatter()

  • 使用plot()绘图时，如果指定样式参数为仅绘制数据点，那么所绘制的就是一幅散列图。但是这种方法所绘制的点无法单独指定颜色和大小。
  • 而scatter()所绘制的散列图却可以指定每个点的颜色和大小。
import numpy as np
import matplotlib.pyplot as plt
plt.plot(np.random.random(100), np.random.random(100), "o")
plt.show()

5. 图像 plt.imread()

import matplotlib.pyplot as plt
import matplotlib.cm as cm
import numpy as np

img = plt.imread("color.png")

# 可以从图像文件读入数据，得到一个表示图像的NumPy数组。
# 它的第一个参数是文件名或文件对象，
# format参数指定图像类型，如果省略，就由文件的扩展名决定图像类型。
# 对于灰度图像，它返回一个形状为(M，N)的数组；
# 对于彩色图像，返回形状为(M，N，C)的数组。
# 其中, M为图像的高度, N为图像的宽度, C为3或4, 表示图像的通道数。

print("img.shape = ", img.shape)
print("img.dtype = ", img.dtype)
plt.imshow(img)

6. 图像 plt.imshow()

  • plt.imshow(z, extent=extent, cmap=cm.seismic, origin="lower”)
# *-* coding:utf8 *-*
import numpy as np
import matplotlib.pyplot as plt
import matplotlib.cm as cm

# ogrid返回的是一组可以用来广播计算的数组
y, x = np.ogrid[-2:2:200j, -2:2:200j]
# 通过数组的广播功能计算出表示函数值的二维数组Z,
# 注意它的第0轴表示Y轴、 第1轴表示X轴。
# 然后将X、Y轴的取值范围保存到extent列表中。
z = x * np.exp(-x**2-y**2)
print("x = ", x)
print("y = ", y)
print("z = ", z)
extent = [np.min(x), np.max(x), np.min(y), np.max(y)]
# 将extent列表传递给 imshow()的extent参数，
# 这样一来，图表的X、Y轴的刻度标签将使用extent列表所指定的范围.

plt.figure(figsize=(8,8))
plt.subplot(221)
plt.imshow(z, extent=extent, origin="lower")
plt.colorbar()
plt.subplot(222)
plt.imshow(z, extent=extent, cmap=cm.Set1, origin="lower")
plt.colorbar()
plt.subplot(223)
plt.imshow(z, extent=extent, cmap=cm.seismic, origin="lower")
plt.colorbar()
plt.show()

# *-* coding:utf8 *-*
import numpy as np
import matplotlib.pyplot as plt

n = 20
Z = np.ones(n)
Z[-1] *= 2
plt.axes([0.025, 0.025, 0.95, 0.95])
# explode:每个扇形离中心的距离;
# colors:设置颜色(每一个都有不同的颜色)这里用的灰度颜色;
plt.pie(Z, explode=Z*0.05, colors=['%f' %(i/float(n)) for i in range(n)] )
# 让他生成一个正圆不是椭圆
plt.gca().set_aspect('equal')
plt.xticks([])
plt.yticks([])
plt.show()
